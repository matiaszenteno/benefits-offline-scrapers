const toKebabCase = (str) => {
  console.log(str);

  return str
    .normalize('NFD')                         // Split letters and diacritics
    .replace(/[\u0300-\u036f]/g, '')          // Remove diacritics
    .replace(/([a-z])([A-Z])/g, '$1-$2')      // Handle camelCase
    .replace(/[\s_]+/g, '-')                  // Replace spaces and underscores with hyphens
    .replace(/[^a-zA-Z0-9-]/g, '')            // Remove other non-alphanumeric characters
    .toLowerCase();
}

window.pressChip = (id) => {
  // null index => press selected chip
  if (!id) {
    id = "";

    document.querySelectorAll("#chips eds-chips button").forEach((item, i) => {
      if (item.children.length > 0) {
        id = item.parentElement.edsChipText;
      }
    });
  }

  const chips = document.querySelectorAll("#chips eds-chips button");
  const event = new Event("mousedown", { bubbles: true, cancelable: true });

  chips.forEach((item) => {
    const label = item.parentElement.edsChipText.toLowerCase();

    if (label == id.toLowerCase()) {
      item.dispatchEvent(event);
      item.click();
    }
  });
};

  let selected = null;

  document.querySelectorAll("#chips eds-chips button").forEach((item) => {
    if (item.children.length > 0) {
      selected = item.parentElement.edsChipText;
    }
  });

  return selected;
};
